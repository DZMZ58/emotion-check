        // Helper function to handle exponential backoff for API calls

        async function exponentialBackoffFetch(url, options, maxRetries = 5, baseDelay = 1000) {

            for (let i = 0; i < maxRetries; i++) {

                try {

                    const response = await fetch(url, options);

                    if (response.ok) {

                        return response;

                    }

                    if (response.status === 429 || response.status >= 500) {

                        // Retry for rate limiting or server errors

                        const delay = baseDelay * Math.pow(2, i) + Math.random() * 1000;

                        console.warn(`Attempt ${i + 1} failed with status ${response.status}. Retrying in ${delay.toFixed(0)}ms...`);

                        await new Promise(resolve => setTimeout(resolve, delay));

                        continue;

                    }

                    // For other client errors (4xx), throw immediately

                    throw new Error(`HTTP error! Status: ${response.status}`);

                } catch (error) {

                    if (i === maxRetries - 1) {

                        throw error;

                    }

                    // For network errors, retry

                    const delay = baseDelay * Math.pow(2, i) + Math.random() * 1000;

                    console.warn(`Attempt ${i + 1} failed with error: ${error.message}. Retrying in ${delay.toFixed(0)}ms...`);

                    await new Promise(resolve => setTimeout(resolve, delay));

                }

            }

            throw new Error("API request failed after all retries.");

        }

                ui.previewImage.style.display = 'block';

                ui.uploadText.textContent = `已选择文件: ${file.name}`;

                ui.analyzeButton.disabled = false;

 

                // Extract Base64 part (data:image/jpeg;base64,...)

                const base64Data = e.target.result.split(',')[1];

                const mimeType = file.type;

                imageBase64 = { data: base64Data, mimeType: mimeType };

        window.analyzeChat = async function() {

                const data = JSON.parse(jsonString);

                    ui.dangerWarning.textContent = warning;

                    ui.warningContainer.style.borderColor = '#22c55e'; // green-500

                    ui.warningContainer.style.backgroundColor = '#f0fdf4'; // green-50

                    ui.dangerWarning.classList.remove('text-red-600');

                    ui.dangerWarning.classList.add('text-green-600');

                    document.querySelector('#warningContainer h3').classList.remove('text-red-600');

                    document.querySelector('#warningContainer h3').classList.add('text-green-600');

                } else {

                    ui.dangerWarning.textContent = warning;

                    ui.warningContainer.style.borderColor = '#ef4444'; // red-500

                    ui.warningContainer.style.backgroundColor = '#fef2f2'; // red-50

                    ui.dangerWarning.classList.remove('text-green-600');

                    ui.dangerWarning.classList.add('text-red-600');

                // 3. 高情商回复

                        replyElement.className = 'p-3 bg-white rounded-lg border-l-4 shadow-sm';

                        let borderColor = '';
